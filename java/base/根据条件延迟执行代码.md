````java
 public void execute(){
        //判断是否具备执行条件
        if (canRun()){
            //执行代码逻辑
            return;
        }
        //不具备条件延迟100ms 再次执行
        new Timer().schedule(new TimerTask() {
            @Override
            public void run() {
                //重复调用自己方法
                execute();
                this.cancel();
            }

        },100);
    }
````
